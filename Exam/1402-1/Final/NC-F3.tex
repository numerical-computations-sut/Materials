	\(\bar{x}\)

	\(Ax = b\)
	و
	\(\bar{x} +d\)
	را جواب دقیق  دستگاه 
	\(Ax = b + r\)
	 در نظر بگیرید. نشان دهید
	\[\frac{\|d\|}{\|\bar{x}\|} \leq K(A) \frac{\|r\|}{\|b\|}.  \]
	\vspace*{.3cm}
	\item[ب)] 
	فرض کنید 
	\(\tilde{x}\)
	تقریبی برای جواب دستگاه خطی
	\(Ax = b\)
	است. قرار دهید:
	\begin{align*}
		& e = x - \tilde{x},\\
		& r= b -A\tilde{x}.
	\end{align*}
	نشان دهید:
	\[
	\frac{1}{K(A)} \frac{\|r\|}{\|b\|} \leq \frac{\|e\|}{\|x\|} \leq  K(A) \frac{\|r\|}{\|b\|}
	\]
		\vspace*{.3cm}
	
\end{enumerate}
